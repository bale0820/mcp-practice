# ── 1. WSL 배포판 확인 및 진입 ──────────────────
wsl -l                        # 설치된 배포판 목록 확인
wsl -d Ubuntu-24.04            # Ubuntu-24.04로 명시적 진입 (기본 Ubuntu와 별개)
wsl --set-default Ubuntu-24.04 # 매번 이걸 기본값으로 쓰고 싶을 때


# ── 2. sudo 비밀번호 재설정 (잊어버렸을 때) ──────
# Windows PowerShell(관리자 권한)에서 실행
# wsl -d Ubuntu-24.04 -u root -- passwd bale


# ── 3. Git + GitHub SSH 연동 ────────────────────
git config --global user.name "본인이름"
git config --global user.email "본인이메일"

# SSH 키 확인 (이미 있으면 재생성 불필요)
ls -la ~/.ssh

# 연결 테스트
ssh -T git@github.com
# → "Hi bale0820! You've successfully authenticated..." 나오면 성공

# remote를 HTTPS → SSH로 변경 (비밀번호 안 물어보게)
git remote set-url origin git@github.com:bale0820/mcp-practice.git
git remote -v   # 확인

# 프로젝트 초기화 + 첫 커밋 + push
mkdir -p ~/projects/mcp-practice && cd ~/projects/mcp-practice
git init
echo "# MCP Practice" > README.md
git add README.md
git commit -m "init"
git branch -M main
git push -u origin main


# ── 4. WSL 안에 MySQL 설치 (실무 표준: 운영환경=리눅스와 통일) ──
sudo apt update
sudo apt install -y mysql-server
sudo service mysql start
sudo service mysql status      # active (running) 확인

# root 인증 방식 설정 + 테스트 DB 생성
sudo mysql

-- MySQL 프롬프트 안에서 실행
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'practice1234';
FLUSH PRIVILEGES;

CREATE DATABASE mcp_practice;
USE mcp_practice;

CREATE TABLE users (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(50),
  email VARCHAR(100)
);

INSERT INTO users (name, email) VALUES
  ('테스트유저1', 'test1@example.com'),
  ('테스트유저2', 'test2@example.com');

EXIT;



# ── 5. MySQL MCP 서버 등록 (실무 표준 패키지: @benborla29/mcp-server-mysql) ──
# 12,205 다운로드, 최근까지 유지보수 중, Claude Code 전용 최적화된 패키지

claude mcp add mcp_server_mysql \
  -e MYSQL_HOST="127.0.0.1" \
  -e MYSQL_PORT="3306" \
  -e MYSQL_USER="root" \
  -e MYSQL_PASS="practice1234" \
  -e MYSQL_DB="mcp_practice" \
  -- npx @benborla29/mcp-server-mysql

claude mcp list   # ✔ Connected 확인

# 기본은 읽기 전용(SELECT만). 쓰기 작업 허용하려면 아래 환경변수 추가:
# -e ALLOW_INSERT_OPERATION="true"
# -e ALLOW_UPDATE_OPERATION="true"
# -e ALLOW_DELETE_OPERATION="true"

# STEP 3: alias 함수 등록 (MySQL과 동일 패턴)

cat >> ~/.bashrc << 'EOF'

mcpgithub-on() {
  claude mcp add github -e GITHUB_PERSONAL_ACCESS_TOKEN="발급받은_토큰" -- npx -y @modelcontextprotocol/server-github
}
mcpgithub-off() {
  claude mcp remove github
}
EOF

source ~/.bashrc
